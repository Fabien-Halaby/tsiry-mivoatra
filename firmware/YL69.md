```cpp
#include <HWCDC.h>
#include <esp32-hal-adc.h>
#include <Arduino.h>
#define SENSOR_PIN 36

// valeurs mesurées avec ta sonde
const int dry = 3100;   // air
const int wet = 250;    // dans l’eau

void setup() {
  Serial.begin(115200);
}

void loop() {
  int raw = analogRead(SENSOR_PIN);
  int percent = map(raw, dry, wet, 0, 100); // 0 % = sec, 100 % = dans l’eau
  percent = constrain(percent, 0, 100);

  Serial.print("Valeur brute : ");
  Serial.print(raw);
  Serial.print("  →  Humidité : ");
  Serial.print(percent);
  Serial.println(" %");

  delay(2000);
}
```
